Frontend - Vue.js + Inertia.js

1. Application Entry Point

The app starts with a Blade template that bootstraps Vue.js.

Root Template (resources/views/app.blade.php)

<!DOCTYPE html>
<html>
<head>
    <title inertia>{{ config('app.name') }}</title>
    
    <!-- Load CSS and JS via Vite -->
    @vite(['resources/js/app.js', "resources/js/Pages/{$page['component']}.vue"])
    
    <!-- Inertia head management -->
    @inertiaHead
</head>
<body class="font-sans antialiased app-background">
    <!-- This is where Vue app mounts -->
    @inertia
</body>
</html>

Vue Initialization (resources/js/app.js)

import { createApp, h } from 'vue';
import { createInertiaApp } from '@inertiajs/vue3';
import { resolvePageComponent } from 'laravel-vite-plugin/inertia-helpers';
import { ZiggyVue } from '../../vendor/tightenco/ziggy';

createInertiaApp({
    // Page title format
    title: (title) => title || 'Leave Request App',
    
    // Dynamically load Vue page components
    resolve: (name) =>
        resolvePageComponent(
            `./Pages/${name}.vue`,  // e.g., ./Pages/Employee/Dashboard.vue
            import.meta.glob('./Pages/**/*.vue')
        ),
    
    // Setup Vue app with plugins
    setup({ el, App, props, plugin }) {
        return createApp({ render: () => h(App, props) })
            .use(plugin)        // Inertia plugin
            .use(ZiggyVue)      // Laravel route() helper
            .mount(el);
    },
});
Key Insight: The resolve function maps Laravel's Inertia::render('Employee/Dashboard') to the Vue file at resources/js/Pages/Employee/Dashboard.vue

2. Layouts - Page Structure

Layouts wrap pages with common elements like navigation and flash messages.

AuthenticatedLayout (for logged-in users)

<script setup>
import { ref, computed } from 'vue';
import { Link, usePage } from '@inertiajs/vue3';
import Dropdown from '@/Components/Dropdown.vue';
import NavLink from '@/Components/NavLink.vue';

// Get shared data from Laravel middleware
const page = usePage();
const user = computed(() => page.props.auth.user);
const flash = computed(() => page.props.flash);

// Determine dashboard route based on role
const dashboardRoute = computed(() => {
    switch (user.value?.role) {
        case 'employee': return 'employee.dashboard';
        case 'dept_manager': return 'manager.dashboard';
        case 'hr_admin': return 'hr.dashboard';
    }
});
</script>

<template>
    <div class="min-h-screen bg-gray-100/85">
        <!-- Navigation Bar -->
        <nav class="border-b border-gray-100 bg-white">
            <NavLink :href="route(dashboardRoute)">Dashboard</NavLink>
            
            <!-- User dropdown with logout -->
            <Dropdown>
                <template #trigger>{{ user.name }}</template>
                <template #content>
                    <DropdownLink :href="route('logout')" method="post">
                        Log Out
                    </DropdownLink>
                </template>
            </Dropdown>
        </nav>

        <!-- Flash Messages -->
        <div v-if="flash?.success" class="bg-green-50 p-4">
            {{ flash.success }}
        </div>

        <!-- Page Header Slot -->
        <header v-if="$slots.header">
            <slot name="header" />
        </header>

        <!-- Main Content Slot -->
        <main>
            <slot />
        </main>
    </div>
</template>

4. Forms - useForm()

Inertia's useForm helper manages form state, submission, and validation errors.

Complete Form Example

<script setup>
import { useForm } from '@inertiajs/vue3';
import PrimaryButton from '@/Components/PrimaryButton.vue';
import TextInput from '@/Components/TextInput.vue';
import InputLabel from '@/Components/InputLabel.vue';
import InputError from '@/Components/InputError.vue';

// Create form with initial values
const form = useForm({
    leave_type: '',
    start_date: '',
    end_date: '',
    reason: '',
});

// Submit function
function submit() {
    // POST to Laravel route
    form.post(route('employee.store-request'), {
        // Reset form on success
        onSuccess: () => form.reset(),
    });
}
</script>

<template>
    <form @submit.prevent="submit">
        <!-- Leave Type -->
        <div>
            <InputLabel value="Leave Type" />
            <select v-model="form.leave_type">
                <option value="">Select...</option>
                <option value="Vacation Leave">Vacation Leave</option>
            </select>
            <!-- Show validation error if exists -->
            <InputError :message="form.errors.leave_type" />
        </div>

        <!-- Start Date -->
        <div>
            <InputLabel value="Start Date" />
            <TextInput type="date" v-model="form.start_date" />
            <InputError :message="form.errors.start_date" />
        </div>

        <!-- Reason -->
        <div>
            <InputLabel value="Reason" />
            <textarea v-model="form.reason"></textarea>
            <InputError :message="form.errors.reason" />
        </div>

        <!-- Submit Button -->
        <PrimaryButton :disabled="form.processing">
            <!-- Show loading state -->
            <span v-if="form.processing">Submitting...</span>
            <span v-else>Submit Request</span>
        </PrimaryButton>
    </form>
</template>

useForm Properties & Methods

Property/Method Purpose
form.field_nameAccess/set form field value
form.errors.field_nameValidation error for field
form.processingTrue while form is submitting
form.post(url)Submit as POST request
form.patch(url)Submit as PATCH request
form.reset()Reset all fields to initial values
form.reset('field')Reset specific field

5. Computed Properties & Reactivity

ref() - Reactive Variables

import { ref } from 'vue';

// Create a reactive variable
const showModal = ref(false);
const selectedType = ref('Vacation Leave');

// Read value
console.log(showModal.value);  // false

// Set value (UI will auto-update)
showModal.value = true;

computed() - Derived Values

import { computed } from 'vue';

const props = defineProps({ balances: Array });

// Computed: auto-updates when balances change
const totalDays = computed(() => {
    return props.balances.reduce((sum, b) => sum + b.balance, 0);
});

// Filter balances with available days
const availableBalances = computed(() => {
    return props.balances.filter(b => b.balance - b.used > 0);
});

watch() - React to Changes

import { watch } from 'vue';

// Watch form.department_id and react when it changes
watch(() => form.department_id, (newValue) => {
    // Clear manager when department changes
    form.manager_id = '';
});

6. Navigation

Link Component (SPA Navigation)

import { Link } from '@inertiajs/vue3';

<!-- Simple link -->
<Link :href="route('employee.dashboard')">Dashboard</Link>

<!-- Link with parameter -->
<Link :href="route('manager.show-request', request.id)">
    View Details
</Link>

<!-- POST link (like logout) -->
<Link :href="route('logout')" method="post" as="button">
    Log Out
</Link>

Programmatic Navigation

import { router } from '@inertiajs/vue3';

// Navigate to a page
router.visit(route('employee.dashboard'));

// Navigate with data preservation
router.get(route('manager.dashboard'), { leave_type: 'Sick Leave' }, {
    preserveState: true,
});

// PATCH request (like canceling a request)
router.patch(route('request.cancel', id));

7. Conditional Rendering

v-if / v-else

<!-- Show only if condition is true -->
<div v-if="requests.data.length > 0">
    <!-- Show table -->
</div>
<div v-else>
    <p>No requests found.</p>
</div>

<!-- Multiple conditions -->
<span v-if="status === 'pending'" class="bg-yellow-100">Pending</span>
<span v-else-if="status === 'hr_approved'" class="bg-green-100">Approved</span>
<span v-else class="bg-red-100">Rejected</span>

Dynamic Classes

<!-- Object syntax -->
<div :class="{ 'bg-green-50': isApproved, 'bg-red-50': isRejected }">

<!-- Array syntax -->
<span :class="['px-2 py-1 rounded-full', getStatusColor(status)]">
